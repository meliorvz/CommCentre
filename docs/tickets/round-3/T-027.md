## Implementation

Create a new page/component as the first step of onboarding:

```tsx
// src/pages/onboarding/WelcomePage.tsx
import { useCompanyProfile } from '@/hooks/useCompanyProfile'; // Hypothetical hook
import { getVerticalConfig } from '@/config/verticals';

export default function WelcomePage() {
    // T-001 adds vertical_id to profile. T-027 needs to use it.
    const { profile } = useCompanyProfile();
    const vertical = getVerticalConfig(profile?.verticalId || 'hospitality');

    return (
        <div className="onboarding-container">
            <h1>{vertical.onboarding.welcomeTitle}</h1>
            <p>{vertical.tagline}</p>
            
            <div className="value-props">
                {vertical.onboarding.valueProps.map((prop, i) => (
                    <ValueProp 
                        key={i}
                        icon={prop.icon} 
                        title={prop.title}
                        description={prop.description}
                    />
                ))}
            </div>
            
            <Button href="/onboarding/profile">Get Started â†’</Button>
        </div>
    );
}
```
