## Background

Track and display onboarding progress so users can resume where they left off.

---

## Implementation

```tsx
// Hook to manage onboarding state
function useOnboardingState() {
    const [state, setState] = useState<OnboardingState | null>(null);
    
    useEffect(() => {
        fetch('/api/onboarding/state').then(r => r.json()).then(setState);
    }, []);
    
    const updateStep = async (step: string, completed: boolean) => {
        await fetch('/api/onboarding/state', {
            method: 'PATCH',
            body: JSON.stringify({ step, completed }),
        });
        setState(prev => ({ ...prev, [step]: completed }));
    };
    
    return { state, updateStep };
}

// Progress indicator
function OnboardingProgress({ currentStep }) {
    const STEPS = ['welcome', 'profile', 'gmail', 'subscription', 'phone', 'telegram', 'complete'];
    const currentIndex = STEPS.indexOf(currentStep);
    
    return (
        <div className="steps">
            {STEPS.map((step, i) => (
                <div key={step} className={i <= currentIndex ? 'complete' : ''}>
                    {i + 1}. {step}
                </div>
            ))}
        </div>
    );
}
```
