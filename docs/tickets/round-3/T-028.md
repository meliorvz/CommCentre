## Background

Enhanced business profile step that collects company information to personalize the AI assistant.

---

## Implementation

```tsx
// Form fields:
// - Company name
// - Website URL
// - Industry/vertical
// - Timezone
// - Assistant name preference

export default function ProfileStep() {
    const [profile, setProfile] = useState({
        companyName: '',
        websiteUrl: '',
        vertical: 'hospitality',
        timezone: Intl.DateTimeFormat().resolvedOptions().timeZone,
        assistantName: 'Mark',
    });
    
    const handleSubmit = async () => {
        await fetch('/api/settings/profile', {
            method: 'POST',
            body: JSON.stringify(profile),
        });
        router.push('/onboarding/gmail');
    };
    
    return (
        <form>
            <Input label="Company Name" value={profile.companyName} ... />
            <Input label="Website URL" value={profile.websiteUrl} ... />
            <Select label="Industry" options={VERTICALS} ... />
            <Select label="Timezone" options={TIMEZONES} ... />
            <Input label="AI Assistant Name" value={profile.assistantName} ... />
            <Button type="submit">Continue</Button>
        </form>
    );
}
```
